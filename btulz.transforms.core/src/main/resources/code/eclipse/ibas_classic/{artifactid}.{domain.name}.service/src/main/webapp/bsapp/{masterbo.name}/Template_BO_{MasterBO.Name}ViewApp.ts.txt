import * as ibas from "../../3rdparty/ibas/index";
import { BORepository${Domain.getName()} } from "../../borep/BORepositories";
import * as bo from "../../borep/bo/index";
import { ${BO.getName()}EditApp } from "./${BO.getName()}EditApp";

/** 查看应用-${BO.getDescription()} */
export class ${BO.getName()}ViewApp extends ibas.BOViewApplication<I${BO.getName()}ViewView> {

    /** 应用标识 */
    static APPLICATION_ID: string = "${ID.newUUID()}";
    /** 应用名称 */
    static APPLICATION_NAME: string = "mu_${Domain.Name().toLowerCase()}_app_${BO.getName().toLowerCase()}_view";

    constructor() {
        super();
        this.id = ${BO.getName()}ViewApp.APPLICATION_ID;
        this.name = ${BO.getName()}ViewApp.APPLICATION_NAME;
        this.description = ibas.i18n.prop(this.name);
    }
    /** 注册视图 */
    protected registerView(): void {
        super.registerView();
        // 其他事件

    }
    /** 视图显示后 */
    protected viewShowed(): void {
        // 视图加载完成
    }
    /** 编辑数据，参数：目标数据 */
    protected editData(): void {
        let app = new ${BO.getName()}EditApp();
        app.navigation = this.navigation;
        app.viewShower = this.viewShower;
        app.run(this.viewData);
    }
    protected viewData: bo.${BO.getName()};
    /** 运行,覆盖原方法 */
    run(...args: any[]): void {
        let data: bo.${BO.getName()} = arguments[0];
        if (ibas.object.isNull(data)) {
            throw new Error(ibas.i18n.prop("msg_invalid_parameter", "view data"));
        }
        this.viewData = data;
        super.run();
    }
}

export interface I${BO.getName()}ViewView extends ibas.IBOViewView {

}
