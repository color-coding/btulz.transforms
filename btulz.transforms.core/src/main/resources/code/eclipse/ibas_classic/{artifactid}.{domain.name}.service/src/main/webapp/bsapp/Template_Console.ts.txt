import * as ibas from "ibas/index";
$BEGIN_BO$
import { ${BO.getName()}Func, ${BO.getName()}ChooseServiceMapping, ${BO.getName()}LinkServiceMapping } from "./${BO.getName().toLowerCase()}/index";
$END_BO$

/** 模块控制台 */
export class Console extends ibas.ModuleConsole {
    /** 模块-标识 */
    static CONSOLE_ID: string = "${ProjectId}";
    /** 模块-名称 */
    static CONSOLE_NAME: string = "${Domain.Name()}";
    /** 根文件名称 */
    static ROOT_FILE_NAME: string = "${Domain.getName().toLowerCase()}/index";
    /** 构造函数 */
    constructor() {
        super();
        this.id = Console.CONSOLE_ID;
        this.name = Console.CONSOLE_NAME;
    }
    private _navigation: ibas.IViewNavigation;
    /** 创建视图导航 */
    navigation(): ibas.IViewNavigation {
        return this._navigation;
    }
    /** 初始化 */
    protected registers(): void {
        // 注册功能
$BEGIN_BO$
        this.register(new ${BO.getName()}Func());
$END_BO$
        // 注册服务应用
$BEGIN_BO$
        this.register(new ${BO.getName()}ChooseServiceMapping());
        this.register(new ${BO.getName()}LinkServiceMapping());
$END_BO$
        // 注册常驻应用

    }
    /** 运行 */
    run(): void {
        // 获取根地址
        let rootUrl: string = ibas.url.rootUrl(Console.ROOT_FILE_NAME);
        // 加载语言-框架默认
        ibas.i18n.load(ibas.strings.format("{0}/resources/languages/${Domain.getName().toLowerCase()}.{1}.json", rootUrl, ibas.i18n.language));
$BEGIN_BO$
        ibas.i18n.load(ibas.strings.format("{0}/resources/languages/bo/${BO.getName().toLowerCase()}.{1}.json", rootUrl, ibas.i18n.language));
$END_BO$
        // 设置资源属性
        this.description = ibas.i18n.prop(this.name.toLowerCase());
        this.icon = ibas.i18n.prop(this.name.toLowerCase() + "_icon");
        // 先加载ui导航
        let uiModules: string[] = [];
        if (!ibas.config.get(ibas.ModuleConsole.CONFIG_ITEM_DISABLE_PLATFORM_VIEW, false)
            && this.plantform === ibas.emPlantform.PHONE) {
            // 使用m类型视图
            uiModules.push("../bsui/m/Navigation");
        } else {
            // 使用c类型视图
            uiModules.push("../bsui/c/Navigation");
        }
        let that: Console = this;
        require(uiModules, function (ui: any): void {
            // 设置导航
            that._navigation = new ui.default();
            // 调用初始化
            that.initialize();
        });
        // 保留基类方法
        super.run();
    }
}
